import React, { useState, useCallback } from 'react';
import { FilterEditor } from 'react-filter-editor';


const options = [
  { name:"name", label:"Name", fieldType: "text" },
  { name:"email", label:"Email", fieldType: "text", icon: "fas fa-envelope" },
  { name:"age", label:"Age", fieldType: "number" },
];

const defaultValues = {
  name: "Juan",
  email: "jpprogrammer1@gmail.com",
  age: "31/05/2000"
}

const App = () => {
  const [values, setValues] = useState({...defaultValues});

   const getTag = useCallback((name, value) => {
    if(name === "name"){
      return "Name Person: " + value;
    }
    if(name === "email"){
      return <><i className="fas fa-at"> {value}</>
    }
  }, []);

  return (
    <FilterEditor options={options} values={values} onChangeValues={setValues} setVisibleValue={getTag}/>
  );
};
