### Import component ðŸ”Ž

```jsx
import React, { useState } from 'react';
import { FilterEditor } from 'react-filter-editor';

const options = [
  { name:"name", label:"Name Person", fieldType: "text" },
  { name:"email", label:"Email", fieldType: "text", icon: "fas fa-envelope" },
  { name:"age", label:"Age", fieldType: "number" },
];

const defaultValues = {
  name: "Juan",
  email: "jpprogrammer1@gmail.com",
  age: "21"
}

const App = () => {
  const [values, setValues] = useState({...defaultValues});
  return (
    <FilterEditor options={options} values={values} onChangeValues={setValues}/>
  );
};
```
